# B-F06 共享脚本模板（完整版）

> **模板 ID**: B-F06
> **版本**: 1.0.0
> **模板名称**: 共享脚本模板（完整版）
> **适用场景**: 完整共享脚本集，覆盖依赖检查、版本同步、构建、测试和发布

---

## 概述

本模板提供 **5 个共享脚本**，覆盖工作区管理的核心场景：

1. **check-deps** - 检查包间依赖方向，防止循环依赖
2. **sync-versions** - 同步工作区中所有包的公共依赖版本
3. **build-all** - 按依赖层级顺序构建所有包
4. **test-all** - 运行所有包的测试，支持过滤
5. **publish** - 统一发布流程

## 脚本 1: 依赖方向检查

```bash
#!/bin/sh
# {{项目名称}} - 依赖方向检查
# 文件路径: scripts/check-deps.sh
# 用途: 检查包间依赖关系，防止循环依赖和违规依赖

set -e

echo "========================================="
echo "  依赖方向检查"
echo "========================================="

# --- 配置 ---
ALLOWED_LAYERS="{{允许的依赖层级}}"
# 示例: "utils:core,components:utils|core,apps:components|utils|core"
# 格式: "包名:允许依赖的包名列表"（用 | 分隔）

# --- 解析依赖关系 ---
echo ""
echo ">>> 解析包间依赖关系..."

PACKAGES=$({{列出所有包命令}})
CYCLES_FOUND=0
VIOLATIONS_FOUND=0

# --- 检查循环依赖 ---
echo ">>> 检查循环依赖..."
for PKG in $PACKAGES; do
  DEPS=$({{获取包依赖命令}} "$PKG")
  for DEP in $DEPS; do
    # 检查是否在工作区包中
    if echo "$PACKAGES" | grep -qw "$DEP"; then
      # 检查反向依赖
      REVERSE_DEPS=$({{获取包依赖命令}} "$DEP")
      if echo "$REVERSE_DEPS" | grep -qw "$PKG"; then
        echo "  [循环依赖] $PKG <-> $DEP"
        CYCLES_FOUND=$((CYCLES_FOUND + 1))
      fi
    fi
  done
done

# --- 检查依赖层级违规 ---
echo ">>> 检查依赖层级..."
for RULE in $(echo "$ALLOWED_LAYERS" | tr ',' '\n'); do
  PKG=$(echo "$RULE" | cut -d: -f1)
  ALLOWED=$(echo "$RULE" | cut -d: -f2 | tr '|' ' ')
  DEPS=$({{获取包依赖命令}} "$PKG")

  for DEP in $DEPS; do
    if echo "$PACKAGES" | grep -qw "$DEP"; then
      if ! echo "$ALLOWED" | grep -qw "$DEP"; then
        echo "  [层级违规] $PKG 不应依赖 $DEP（允许: $ALLOWED）"
        VIOLATIONS_FOUND=$((VIOLATIONS_FOUND + 1))
      fi
    fi
  done
done

# --- 输出结果 ---
echo ""
if [ $CYCLES_FOUND -eq 0 ] && [ $VIOLATIONS_FOUND -eq 0 ]; then
  echo "依赖方向检查通过"
else
  echo "发现 $CYCLES_FOUND 个循环依赖, $VIOLATIONS_FOUND 个层级违规"
  exit 1
fi
```

## 脚本 2: 版本同步

```bash
#!/bin/sh
# {{项目名称}} - 版本同步
# 文件路径: scripts/sync-versions.sh
# 用途: 同步工作区中所有包的公共依赖版本，确保版本一致

set -e

echo "========================================="
echo "  版本同步"
echo "========================================="

# --- 配置 ---
SYNC_DEPS="{{需要同步的依赖列表}}"
# 示例: "framework,utils-lib,types-package"
SYNC_DEV_DEPS="{{需要同步的开发依赖列表}}"
# 示例: "test-runner,build-tool,linter"

# --- 收集版本信息 ---
echo ""
echo ">>> 收集各包的依赖版本..."

PACKAGES=$({{列出所有包命令}})

# 为每个依赖收集所有包中使用的版本
for DEP in $SYNC_DEPS $SYNC_DEV_DEPS; do
  echo ""
  echo "--- $DEP ---"
  VERSIONS=$({{收集依赖版本命令}} "$DEP")
  UNIQUE_VERSIONS=$(echo "$VERSIONS" | sort -u | grep -v '^$')

  VERSION_COUNT=$(echo "$UNIQUE_VERSIONS" | wc -l)

  if [ "$VERSION_COUNT" -eq 1 ]; then
    echo "  所有包使用相同版本: $UNIQUE_VERSIONS"
  else
    echo "  [不一致] 发现多个版本:"
    echo "$UNIQUE_VERSIONS" | while read -r VER; do
      echo "    - $VER"
    done
    echo "  建议统一为最新版本"
  fi
done

# --- 自动同步（可选） ---
if [ "${{自动同步}}" = "true" ]; then
  echo ""
  echo ">>> 自动同步版本..."
  for DEP in $SYNC_DEPS; do
    LATEST_VERSION=$({{获取最新版本命令}} "$DEP")
    echo "  同步 $DEP → $LATEST_VERSION"
    {{更新依赖版本命令}} "$DEP" "$LATEST_VERSION"
  done
  echo "版本同步完成，请检查变更后提交"
fi
```

## 脚本 3: 全量构建

```bash
#!/bin/sh
# {{项目名称}} - 全量构建
# 文件路径: scripts/build-all.sh
# 用途: 按依赖层级顺序构建所有包

set -e

echo "========================================="
echo "  全量构建"
echo "========================================="

# --- 配置 ---
BUILD_TARGET="${1:-all}"  # 默认构建所有包
PARALLEL="${{并行构建}}"   # true/false

# --- 检测包管理器 ---
PKG_MANAGER="{{包管理器命令}}"

# --- 安装依赖 ---
echo ""
echo ">>> 安装依赖..."
$PKG_MANAGER {{安装命令}}

# --- 按层级构建 ---
echo ""
echo ">>> 按依赖层级构建..."

if [ "$BUILD_TARGET" = "all" ]; then
  # 获取依赖层级
  LEVELS=$({{获取依赖层级命令}})

  for LEVEL in $LEVELS; do
    echo ""
    echo "--- 层级 $LEVEL ---"
    PACKAGES_AT_LEVEL=$({{获取层级包命令}} "$LEVEL")

    if [ "$PARALLEL" = "true" ]; then
      # 并行构建同一层级的包
      $PKG_MANAGER {{构建命令}} --filter "$PACKAGES_AT_LEVEL" --parallel
    else
      # 串行构建
      for PKG in $PACKAGES_AT_LEVEL; do
        echo "  构建: $PKG"
        $PKG_MANAGER {{构建命令}} --filter "$PKG"
      done
    fi
  done
else
  # 构建指定包及其依赖
  echo "  构建: $BUILD_TARGET"
  $PKG_MANAGER {{构建命令}} --filter "$BUILD_TARGET..."
fi

echo ""
echo "========================================="
echo "  构建完成"
echo "========================================="
```

## 脚本 4: 全量测试

```bash
#!/bin/sh
# {{项目名称}} - 全量测试
# 文件路径: scripts/test-all.sh
# 用途: 运行所有包的测试，支持按包名或层级过滤

set -e

echo "========================================="
echo "  全量测试"
echo "========================================="

# --- 配置 ---
FILTER="${1:-all}"       # 过滤条件: 包名 / 层级 / all
COVERAGE="${{生成覆盖率}}" # true/false
WATCH_MODE="${{监听模式}}" # true/false

# --- 检测包管理器 ---
PKG_MANAGER="{{包管理器命令}}"

# --- 安装依赖 ---
echo ""
echo ">>> 安装依赖..."
$PKG_MANAGER {{安装命令}}

# --- 运行测试 ---
echo ""
echo ">>> 运行测试..."

TEST_CMD="{{测试命令}}"

if [ "$COVERAGE" = "true" ]; then
  TEST_CMD="$TEST_CMD --coverage"
fi

if [ "$WATCH_MODE" = "true" ]; then
  TEST_CMD="$TEST_CMD --watch"
fi

case "$FILTER" in
  all)
    echo "运行所有包的测试..."
    $PKG_MANAGER run $TEST_CMD --filter "{{包过滤模式}}"
    ;;
  changed)
    echo "仅运行变更包的测试..."
    CHANGED=$({{获取变更包命令}})
    if [ -n "$CHANGED" ]; then
      $PKG_MANAGER run $TEST_CMD --filter "$CHANGED"
    else
      echo "没有变更的包，跳过测试"
    fi
    ;;
  *)
    echo "运行指定包的测试: $FILTER"
    $PKG_MANAGER run $TEST_CMD --filter "$FILTER"
    ;;
esac

echo ""
echo "========================================="
echo "  测试完成"
echo "========================================="
```

## 脚本 5: 统一发布

```bash
#!/bin/sh
# {{项目名称}} - 统一发布
# 文件路径: scripts/publish.sh
# 用途: 统一发布工作区中的包，按依赖顺序发布

set -e

echo "========================================="
echo "  统一发布"
echo "========================================="

# --- 配置 ---
REGISTRY="{{包注册表地址}}"
DRY_RUN="${{试运行}}"   # true/false
BUMP_TYPE="${1:-patch}" # patch / minor / major

# --- 前置检查 ---
echo ""
echo ">>> 前置检查..."

# 检查是否在主分支
BRANCH=$(git rev-parse --abbrev-ref HEAD)
if [ "$BRANCH" != "{{默认分支}}" ]; then
  echo "错误: 发布必须在 {{默认分支}} 分支上进行（当前: $BRANCH）"
  exit 1
fi

# 检查是否有未提交的变更
if [ -n "$(git status --porcelain)" ]; then
  echo "错误: 存在未提交的变更，请先提交或暂存"
  exit 1
fi

# 检查是否已登录注册表
echo ">>> 验证注册表认证..."
$PKG_MANAGER {{验证认证命令}} "$REGISTRY"

# --- 确定发布范围 ---
echo ""
echo ">>> 确定需要发布的包..."
CHANGED_PACKAGES=$({{获取变更包命令}})

if [ -z "$CHANGED_PACKAGES" ]; then
  echo "没有需要发布的包"
  exit 0
fi

echo "以下包将被发布:"
echo "$CHANGED_PACKAGES"

# --- 按依赖顺序发布 ---
echo ""
echo ">>> 按依赖顺序发布..."

PACKAGES=$({{获取依赖层级命令}} --reverse)

for PKG in $PACKAGES; do
  # 仅发布有变更的包
  if ! echo "$CHANGED_PACKAGES" | grep -qw "$PKG"; then
    continue
  fi

  echo ""
  echo "--- 发布: $PKG ---"

  # 更新版本号
  CURRENT_VERSION=$({{获取包版本命令}} "$PKG")
  NEW_VERSION=$({{计算新版本命令}} "$CURRENT_VERSION" "$BUMP_TYPE")
  echo "  版本: $CURRENT_VERSION → $NEW_VERSION"
  {{更新包版本命令}} "$PKG" "$NEW_VERSION"

  # 构建
  echo "  构建中..."
  $PKG_MANAGER {{构建命令}} --filter "$PKG"

  # 发布
  if [ "$DRY_RUN" = "true" ]; then
    echo "  [试运行] 跳过实际发布"
  else
    echo "  发布中..."
    $PKG_MANAGER {{发布命令}} --filter "$PKG" --registry "$REGISTRY"
    echo "  发布成功: $PKG@$NEW_VERSION"
  fi
done

# --- 提交版本变更 ---
if [ "$DRY_RUN" != "true" ]; then
  echo ""
  echo ">>> 提交版本变更..."
  git add -A
  git commit -m "chore: 发布版本 $BUMP_TYPE"
  git push origin "{{默认分支}}"
fi

echo ""
echo "========================================="
echo "  发布完成"
echo "========================================="
```

## 脚本关系

```
check-deps ──→ 验证依赖方向
sync-versions ──→ 同步依赖版本
build-all ──→ 按序构建
test-all ──→ 运行测试
publish ──→ 统一发布
```

## 使用方式

```bash
# 赋予执行权限
chmod +x scripts/*.sh

# 依赖检查
./scripts/check-deps.sh

# 版本同步
./scripts/sync-versions.sh

# 全量构建
./scripts/build-all.sh           # 构建所有包
./scripts/build-all.sh my-pkg    # 构建指定包

# 全量测试
./scripts/test-all.sh            # 测试所有包
./scripts/test-all.sh changed    # 仅测试变更包
./scripts/test-all.sh my-pkg     # 测试指定包

# 统一发布
./scripts/publish.sh patch       # 补丁版本发布
./scripts/publish.sh minor       # 次版本发布
./scripts/publish.sh major       # 主版本发布
```

## 适配指南

- 根据实际使用的包管理器调整命令语法。
- 依赖层级检测可通过包管理器的内置命令或自定义脚本实现。
- 版本同步功能可结合工作区目录协议（catalog protocol）实现自动化。

---

## 注意事项

1. 发布脚本应仅在主分支上运行，避免误发布开发中的代码。
2. 试运行模式（dry-run）可用于验证发布流程而不实际执行。
3. 依赖方向检查建议集成到 CI 流水线中，作为合并前的强制检查。
4. 版本同步前建议先运行依赖检查，确保依赖关系正确。
